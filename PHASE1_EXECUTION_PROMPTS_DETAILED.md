## PROMPT 1: Create `invoice_billables` Migration

**File:** `supabase/migrations/20260325_phase1_invoice_billables.sql`

**Context:** 
- Dashboard currently shows "Ready to Bill" card pulling from `billing_candidates` view
- We need to store PM-entered manual amounts for lump-sum phases before invoice generation
- This table will be the source of truth for "what to bill" for each phase each month

**Cursor Prompt:**

```
Create a new Supabase migration file at:
supabase/migrations/20260325_phase1_invoice_billables.sql

This table stores the billing amounts that will be used for invoice generation.
PMs enter manual amounts for lump-sum phases; hourly phases are auto-calculated.

CREATE TABLE IF NOT EXISTS public.invoice_billables (
  id BIGINT PRIMARY KEY GENERATED BY IDENTITY,
  project_id BIGINT NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
  contract_phase_id BIGINT NOT NULL REFERENCES public.contract_phases(id) ON DELETE CASCADE,
  billing_period TEXT NOT NULL, -- format: "2026-03"
  
  -- For hourly phases: calculated value (hours × billable_rate)
  calculated_billable NUMERIC(14,2), 
  
  -- For lump-sum phases: PM manually enters this value
  -- For hourly: if PM wants to override, they can set this
  manual_override_amount NUMERIC(14,2),
  
  -- Final amount that will be billed (either manual_override or calculated)
  final_billable_amount NUMERIC(14,2) GENERATED ALWAYS AS (
    COALESCE(manual_override_amount, calculated_billable)
  ) STORED,
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  
  UNIQUE(project_id, contract_phase_id, billing_period)
);

-- Indexes for query performance
CREATE INDEX IF NOT EXISTS ix_invoice_billables_project_id 
  ON public.invoice_billables(project_id);
CREATE INDEX IF NOT EXISTS ix_invoice_billables_contract_phase_id 
  ON public.invoice_billables(contract_phase_id);
CREATE INDEX IF NOT EXISTS ix_invoice_billables_period 
  ON public.invoice_billables(billing_period);
CREATE INDEX IF NOT EXISTS ix_invoice_billables_project_period 
  ON public.invoice_billables(project_id, billing_period);

-- Enable RLS
ALTER TABLE public.invoice_billables ENABLE ROW LEVEL SECURITY;

-- RLS Policy: Admin full access
CREATE POLICY "invoice_billables_admin_full" ON public.invoice_billables
  AS PERMISSIVE FOR ALL USING (
    (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'admin'
  );

-- RLS Policy: Project Managers can manage billables for their projects
CREATE POLICY "invoice_billables_pm_own_projects" ON public.invoice_billables
  AS PERMISSIVE FOR ALL USING (
    project_id IN (
      SELECT p.id FROM public.projects p
      WHERE p.pm_id = auth.uid()
        OR p.id IN (
          SELECT project_id FROM public.project_team_assignments
          WHERE user_id = auth.uid()
        )
    )
  );

Return only the migration SQL content, with no commentary.
```

**Validation:**
- [ ] Table created with correct schema
- [ ] Unique constraint on (project_id, contract_phase_id, billing_period)
- [ ] Generated column works correctly
- [ ] RLS policies enable admin and PM access
- [ ] Indexes exist for query performance

## PROMPT 2: Create `project_team_assignments` Table

**File:** `supabase/migrations/20260325_phase1_project_team_assignments.sql`

**Context:**
- Currently only `projects.pm_id` tracks who's responsible for a project
- We need a way to assign multiple team members to a project
- This will be used to filter dashboard data per user + enable PM billables interface

**Cursor Prompt:**

```
Create a new Supabase migration file at:
supabase/migrations/20260325_phase1_project_team_assignments.sql

This table tracks which users are assigned to which projects (PMs + team members).

CREATE TABLE IF NOT EXISTS public.project_team_assignments (
  id BIGINT PRIMARY KEY GENERATED BY IDENTITY,
  project_id BIGINT NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  role TEXT NOT NULL DEFAULT 'team_member', -- 'project_manager' or 'team_member'
  assigned_at TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  
  UNIQUE(project_id, user_id)
);

CREATE INDEX IF NOT EXISTS ix_project_team_project_id 
  ON public.project_team_assignments(project_id);
CREATE INDEX IF NOT EXISTS ix_project_team_user_id 
  ON public.project_team_assignments(user_id);
CREATE INDEX IF NOT EXISTS ix_project_team_user_role 
  ON public.project_team_assignments(user_id, role);

ALTER TABLE public.project_team_assignments ENABLE ROW LEVEL SECURITY;

-- Admin full access
CREATE POLICY "project_team_admin_full" ON public.project_team_assignments
  AS PERMISSIVE FOR ALL USING (
    (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'admin'
  );

-- Users can see assignments for projects they're on
CREATE POLICY "project_team_see_own" ON public.project_team_assignments
  AS PERMISSIVE FOR SELECT USING (
    project_id IN (
      SELECT id FROM public.projects WHERE pm_id = auth.uid()
      UNION
      SELECT project_id FROM public.project_team_assignments WHERE user_id = auth.uid()
    )
  );

-- PMs and admin can modify
CREATE POLICY "project_team_pm_modify" ON public.project_team_assignments
  AS PERMISSIVE FOR INSERT WITH CHECK (
    project_id IN (
      SELECT id FROM public.projects WHERE pm_id = auth.uid()
    )
    OR (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'admin'
  );

CREATE POLICY "project_team_pm_update" ON public.project_team_assignments
  AS PERMISSIVE FOR UPDATE USING (
    project_id IN (
      SELECT id FROM public.projects WHERE pm_id = auth.uid()
    )
    OR (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'admin'
  );

CREATE POLICY "project_team_pm_delete" ON public.project_team_assignments
  AS PERMISSIVE FOR DELETE USING (
    project_id IN (
      SELECT id FROM public.projects WHERE pm_id = auth.uid()
    )
    OR (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'admin'
  );

Return only the migration SQL, no commentary.
```

const { data: monthlyMultipliers, isLoading: loadingMultipliers } = useQuery({
  queryKey: ['dashboard-monthly-multipliers'],
  queryFn: async () => {
    // Get last 12 months of data
    const currentMonthStart = new Date()
    currentMonthStart.setDate(1)
    const firstMonth = new Date(currentMonthStart.getFullYear(), currentMonthStart.getMonth() - 12, 1)
    const sinceDate = firstMonth.toISOString().slice(0, 10)
    const currentMonthStartDate = currentMonthStart.toISOString().slice(0, 10)
    
    // Helper to format month key
    const monthKey = (date: Date) => `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}`
    const formatMonthLabel = (key: string) => {
      const [year, month] = key.split('-')
      const date = new Date(Number(year), Number(month) - 1, 1)
      const shortMonth = date.toLocaleString('en-US', { month: 'short' })
      return `${shortMonth} '${String(year).slice(-2)}`
    }
    
    const months = Array.from({ length: 12 }, (_, index) => {
      const monthDate = new Date(firstMonth.getFullYear(), firstMonth.getMonth() + index, 1)
      return monthKey(monthDate)
    })
    
    // Fetch all contract_phases to identify C* phases
    const { data: allPhases } = await supabase
      .from('contract_phases')
      .select('id, project_id, phase_code')
    const cPhaseIds = (allPhases || [])
      .filter(p => (p.phase_code || '').startsWith('C'))
      .map(p => p.id)
    
    // Fetch invoices for C* phases
    const { data: cInvoiceLines } = await supabase
      .from('invoice_line_items')
      .select('amount, created_at')
      .in('contract_phase_id', cPhaseIds)
      .gte('created_at', sinceDate)
      .lt('created_at', currentMonthStartDate)
    
    // Fetch time entries for C* phases to calculate labor cost
    const { data: cTimeEntries } = await supabase
      .from('time_entries')
      .select('id, hours, entry_date')
      .in('contract_phase_id', cPhaseIds)
      .gte('entry_date', sinceDate)
      .lt('entry_date', currentMonthStartDate)
    
    // Get billable rates for time entries
    let rates = new Map<number, number>()
    if (cTimeEntries && cTimeEntries.length > 0) {
      const entryIds = cTimeEntries.map(e => e.id)
      const { data: rateSnapshots } = await supabase
        .from('time_entry_bill_rates')
        .select('time_entry_id, resolved_hourly_rate')
        .in('time_entry_id', entryIds)
      (rateSnapshots || []).forEach(r => {
        rates.set(r.time_entry_id, Number(r.resolved_hourly_rate) || 0)
      })
    }
    
    // Calculate revenue and cost by month
    const revenueBuckets = new Map<string, number>()
    const costBuckets = new Map<string, number>()
    
    (cInvoiceLines || []).forEach(line => {
      const createdMonth = (line.created_at || '').slice(0, 7)
      if (!createdMonth) return
      const serviceMonth = new Date(`${createdMonth}-01T00:00:00`)
      const key = monthKey(new Date(serviceMonth.getFullYear(), serviceMonth.getMonth() - 1, 1))
      revenueBuckets.set(key, (revenueBuckets.get(key) || 0) + (Number(line.amount) || 0))
    })
    
    (cTimeEntries || []).forEach(entry => {
      const entryMonth = (entry.entry_date || '').slice(0, 7)
      if (!entryMonth) return
      const rate = rates.get(entry.id) || 0
      const cost = (Number(entry.hours) || 0) * rate
      costBuckets.set(entryMonth, (costBuckets.get(entryMonth) || 0) + cost)
    })
    
    // Calculate multiplier for each month
    return months.map(month => {
      const revenue = revenueBuckets.get(month) || 0
      const cost = costBuckets.get(month) || 0
      const multiplier = cost > 0 ? revenue / cost : 0
      return {
        month,
        monthLabel: formatMonthLabel(month),
        multiplier: Math.round(multiplier * 100) / 100, // Round to 2 decimals
      }
    })
  },
})
